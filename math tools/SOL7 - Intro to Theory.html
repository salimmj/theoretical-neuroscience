from scipy.integrate import solve_ivp
from functools import partial

def ode_step(t, x, g, J, I, R0, Rmax, tau):
    
    phi = np.zeros(x.shape)
    phi[x<=0] = R0*np.tanh(x[x<=0]/R0)
    phi[x>0]  = (Rmax-R0)*np.tanh(x[x>0]/(Rmax-R0))

    r    = R0 + phi
        
    dxdt = 1/tau*(-x + g*(J@r) + I[:, int(np.floor(t-1e-5))])
    
    return dxdt





def larry_model(parameters):
    ## Parameters

    N     = parameters['N']
    tau   = parameters['tau']*parameters['fs']
    T     = parameters['T']*parameters['fs']
    g     = parameters['g']
    J     = parameters['J']
    inp   = parameters['inp']
    Rmax  = parameters['Rmax']
    R0    = parameters['R0']
    inp_t = parameters['inp_t']
    spon  = parameters['spon']*parameters['fs']



    ## Input current
    
    I_zero              = np.zeros((N,T))
    I_step              = np.tile(np.concatenate((np.zeros((1, spon)), inp*np.ones((1, T - spon))),axis=1), (N,1))
    I_const             = inp*np.ones((N, T))
    

    if inp_t is 'zero':
        I = I_zero
    elif inp_t is 'const':
        I = I_const
    elif inp_t is 'step':
        I = I_step


    ## Initialize

    x0      = np.random.rand(N, 1)


    ## Solving equations
    sol = solve_ivp(partial(ode_step,g=g,J=J,I=I,R0=R0,Rmax=Rmax,tau=tau),[0,T],x0.squeeze())


    x         = sol.y
    t         = sol.t
    phi       = np.zeros(x.shape)
    phi[x<=0] = R0*np.tanh(x[x<=0]/R0)
    phi[x>0]  = (Rmax-R0)*np.tanh(x[x>0]/(Rmax-R0))

    r         = (R0 + phi).T

    t = t / parameters['fs']

    return r, t, I

    parameters['J'] = np.random.normal(loc=0.0, scale=1/np.sqrt(parameters['N']), size=(parameters['N'], parameters['N'])) # connection matrix
    parameters['inp'] = inp_coeff_range[i]*parameters['inp_h']
    
    r, t, I = larry_model(parameters)
    
    
    plt.plot(t,r[:,0:10]/parameters['Rmax'])
    plt.title('I = ' + str(parameters['inp']))
    plt.show()
    
    pca = PCA()
    plt.subplot(1,2,1)
    pca.fit_transform(r[t<parameters['spon'],0:10]/parameters['Rmax'])
    plt.plot(np.arange(len(pca.explained_variance_)), pca.explained_variance_)
    plt.title('Spontaneous')
    
    pca = PCA()
    plt.subplot(1,2,2)
    pca.fit_transform(r[t>=parameters['spon'],0:10]/parameters['Rmax'])
    plt.plot(np.arange(len(pca.explained_variance_)), pca.explained_variance_)
    plt.title('Evoked')
    plt.show()
    
# Stimulus seems to be quenching the variability

Problem 2: High Dimensional Spaces¶

2a) Angle between two random vectors in N-dim space as a function of N

Note that the range of angles is 0 to 90 degrees because of symmetry in the cosine function. Makes sense that the average angle in the 2D case (vectors lie within the unit circle) is 45 degrees. Randomly drawn vectors are more likely to be orthogonal in high dimensional spaces. An argument can be formulated based on the Law of Large Numbers and Central Limit Theorem.

In [8]:
trials = 100
N = np.linspace(1,50,50)
phi = np.zeros((len(N), trials))

for n in range(len(N)):
    for t in range(trials):
        x = np.random.randn(np.int(N[n]))
        y = np.random.randn(np.int(N[n]))
        phi_prime = np.arccos(np.dot(x,y)/(np.linalg.norm(x)*np.linalg.norm(y)))
        phi[n, t] = min(phi_prime, np.pi - phi_prime)

phi_mean = np.mean(phi, 1)

plt.figure()
plt.plot(N, np.rad2deg(phi_mean), 'o-')
plt.xlabel('N')
plt.ylim([-5, 90])
plt.ylabel('angle (degrees)')
plt.show()

2b) Probability of random point in an N-dim unit cube lies within the N-dim unit sphere.

On a sphere, for the unit vector in the (1,1,...,1) direction, the component of the sphere along each of the N axes is $\frac{1}{\sqrt(N)}$. For 2 dimensions, it's 1/sqrt(2) along both the x and y directions. So as N goes to infinite, the probability of a point lying within the sphere is the probability of that point lying on any given axis. The sphere becomes "spikey" in high dimensions. You can do a similar thought experiment for a cube. The distance from the vertex of the cube to the origin is sqrt(N). So as N goes to infinite, the vertices of the cube become more distant from the origin. Instead of Euclidean distance, we can use other norms like L1 norm and explore how the geometry changes in high dimensional spaces.

N = np.linspace(1,50,50)
trials = 10000 # number of points to sample 
frac = np.zeros(len(N)) # fraction of points lying within a uniform sphere
# generate random numbers from -1 to 1 drawing from a uniform distrbution
for n in range(5):
    x = 2*np.random.rand(np.int(N[n]), trials)-1 # draw 100 points randomly within an N-dim cube
    is_within_sphere = np.sum(x**2, 0) < 1  # is the point within a unit sphere? Is the distance from the N-dim point to the origin less than 1?
    frac[n] = np.sum(is_within_sphere)/trials
                         
plt.figure()
plt.plot(N, frac, 'o-')
plt.ylabel('Probability')
plt.xlabel('N')
